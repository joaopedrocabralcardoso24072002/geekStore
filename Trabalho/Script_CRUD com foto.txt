    public partial class FrmProduto : MetroFramework.Forms.MetroForm
    {
        public FrmProduto()
        {
            InitializeComponent();
        }

        private void btnExcluir_Click(object sender, EventArgs e)
        {
            try
            {
                if (pbxFoto.Image != null)
                {
                    pbxFoto.Image.Dispose();
                    pbxFoto.Image = null;
                }

                int Id = Convert.ToInt32(txtId.Text.Trim());

                Produto produto = new Produto();
                produto.Localizar(Id);

                string nome = produto.nome;
                string foto = nome.Replace(" ", "");

                System.IO.FileInfo fi = new System.IO.FileInfo(@"C:\Programas\LojaGeek\Clientes\" + foto + ".jpg");
                fi.Delete();

                produto.Excluir(Id);

                MessageBox.Show("Produto exclu√≠do com sucesso!", "Sucesso", MessageBoxButtons.OK, MessageBoxIcon.Information);
                
		List<Produto> produtos = produto.listaprodutos();
                dgvProduto.DataSource = produtos;

                txtNome.Text = "";
                CbxTipo.Text = "";
                txtQuantidade.Text = "";
                txtPreco.Text = "";
                txtId.Text = "";
                pbxFoto.Image = null;
                pbxFoto.Update();

                this.ActiveControl = txtNome;
            }
            catch (Exception er)
            {
                MessageBox.Show(er.Message, "Erro", MessageBoxButtons.OK, MessageBoxIcon.Error);
            }
        }
    }
}
